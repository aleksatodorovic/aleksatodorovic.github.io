I figured I'd be doing a lot of line plotting, specifically of just arms deals by buyer/seller over a time span so I've included these two functions as baby steps towards automating some visualizations/analyses.¶

In [130]:
def plot_sellers(countries): #input should be a list
    fig, ax = subplots(figsize=(20,10))
    for x in countries:
        arms.loc[(arms['order_year']>1949) & (arms['seller']==x)]['order_year'].groupby(arms.order_year).agg('count').plot(kind='line',ax=ax)
    plt.suptitle('Arms Transfers by Seller', x=0.5, y=0.95, ha='center', fontsize='xx-large')
    fig.text(0.5, 0.04, 'Year', ha='center')
    fig.text(0.09, 0.5, 'Arms Deals', va='center', rotation='vertical')
    ax.legend(countries)
    return ax

def plot_buyers(countries): #input should be a list
    fig, ax = subplots(figsize=(20,10))
    for x in countries:
        arms.loc[(arms['order_year']>1949) & (arms['buyer']==x)]['order_year'].groupby(arms.order_year).agg('count').plot(kind='line',ax=ax)
    plt.suptitle('Arms Transfers by Buyer', x=0.5, y=0.95, ha='center', fontsize='xx-large')
    fig.text(0.5, 0.04, 'Year', ha='center')
    fig.text(0.09, 0.5, 'Arms Deals', va='center', rotation='vertical')
    ax.legend(countries)
    return ax
